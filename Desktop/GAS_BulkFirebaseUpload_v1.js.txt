// ── HELPERS ────────────────────────────────────────────────────
function _ts(){ return new Date().toISOString(); }
function _log(msg){ Logger.log('[BulkUpload] ' + _ts() + ' | ' + msg); }
function _setProp(k,v){ PropertiesService.getScriptProperties().setProperty(k,String(v)); }
function _getProp(k){ return PropertiesService.getScriptProperties().getProperty(k)||null; }

function _getToken(){
  var now=Math.floor(Date.now()/1000);
  var h=Utilities.base64EncodeWebSafe(JSON.stringify({alg:'RS256',typ:'JWT'}));
  var p=Utilities.base64EncodeWebSafe(JSON.stringify({
    iss:FS_CLIENT_EMAIL,sub:FS_CLIENT_EMAIL,
    aud:'https://oauth2.googleapis.com/token',
    iat:now,exp:now+3600,
    scope:'https://www.googleapis.com/auth/datastore'}));
  var sig=Utilities.base64EncodeWebSafe(Utilities.computeRsaSha256Signature(h+'.'+p,FS_PRIVATE_KEY));
  var resp=UrlFetchApp.fetch('https://oauth2.googleapis.com/token',{
    method:'post',contentType:'application/x-www-form-urlencoded',
    payload:'grant_type=urn%3Aietf%3Aparams%3Aoauth%3Agrant-type%3Ajwt-bearer&assertion='+h+'.'+p+'.'+sig,
    muteHttpExceptions:true});
  return JSON.parse(resp.getContentText()).access_token||null;
}

function _fsWriteChunk(docPath, cab, rows, chunkIdx, totalChunks, token) {
  // Usa um subdocumento por chunk: snapshots_bulk/NOME/chunks/0001
  var url = 'https://firestore.googleapis.com/v1/projects/' + FIRESTORE_PROJECT_ID +
            '/databases/(default)/documents/' + docPath;
  var payload = {
    fields: {
      cab:         { stringValue: JSON.stringify(cab) },
      rows:        { stringValue: JSON.stringify(rows) },
      chunkIdx:    { integerValue: String(chunkIdx) },
      totalChunks: { integerValue: String(totalChunks) },
      updatedAt:   { stringValue: _ts() },
      nLinhas:     { integerValue: String(rows.length) }
    }
  };
  var r = UrlFetchApp.fetch(url, {
    method: 'patch', contentType: 'application/json',
    headers: { Authorization: 'Bearer ' + token },
    payload: JSON.stringify(payload), muteHttpExceptions: true
  });
  var code = r.getResponseCode();
  if (code !== 200 && code !== 201) {
    _log('ERRO Firestore ' + code + ': ' + r.getContentText().substring(0,200));
    return false;
  }
  return true;
}

function _parseLinhas(rows) {
  var out = [];
  for (var r = 0; r < rows.length; r++) {
    var l = rows[r], ok = false;
    for (var c = 0; c < l.length; c++) if (l[c] !== '' && l[c] !== null) { ok = true; break; }
    if (!ok) continue;
    out.push(l.map(function(v) {
      if (!v && v !== 0) return '';
      if (v instanceof Date) return Utilities.formatDate(v, Session.getScriptTimeZone(), 'dd/MM/yyyy HH:mm:ss');
      return String(v);
    }));
  }
  return out;
}

// ── LISTAR TODAS AS PLANILHAS DA PASTA ─────────────────────────
function listarTodasPlanilhas() {
  var pasta = DriveApp.getFolderById(PASTA_ID);
  var files = pasta.getFiles();
  var lista = [];
  while (files.hasNext()) {
    var f = files.next();
    lista.push({ nome: f.getName(), id: f.getId(), tipo: f.getMimeType() });
  }
  lista.sort(function(a,b) { return a.nome.localeCompare(b.nome); });
  _log('Total de arquivos na pasta: ' + lista.length);
  lista.forEach(function(f) { _log('  ' + f.nome + ' (' + f.id + ')'); });
  return lista;
}

// ── INICIAR UPLOAD TOTAL ────────────────────────────────────────
// Execute uma vez. Instala trigger de 1min que processa chunk a chunk.
function iniciarUploadTotal() {
  // Remove trigger anterior se existir
  ScriptApp.getProjectTriggers().forEach(function(t) {
    if (t.getHandlerFunction() === 'processarChunkBulk') ScriptApp.deleteTrigger(t);
  });

  // Lista todas as planilhas e serializa o índice
  var lista = listarTodasPlanilhas().filter(function(f) {
    return f.tipo === 'application/vnd.google-apps.spreadsheet';
  });

  if (!lista.length) { _log('❌ Nenhuma planilha encontrada na pasta.'); return; }

  _setProp('bulk_lista',   JSON.stringify(lista));
  _setProp('bulk_planIdx', '0');
  _setProp('bulk_fromRow', '2');
  _setProp('bulk_chunkGlobal', '0');
  _setProp('bulk_totalLinhas', '0');
  _setProp('bulk_ativo', '1');

  _log('🚀 Upload total iniciado! ' + lista.length + ' planilhas.');
  _log('Instalar trigger de 1min → processarChunkBulk()');

  ScriptApp.newTrigger('processarChunkBulk').timeBased().everyMinutes(1).create();
  _log('Trigger instalado. Acompanhe pelo log de execução.');
  _log('Execute verificarStatusUpload() a qualquer momento para ver o progresso.');
}

// ── PROCESSAR UM CHUNK (chamado pelo trigger) ───────────────────
function processarChunkBulk() {
  if (_getProp('bulk_ativo') !== '1') {
    // Desativa o trigger
    ScriptApp.getProjectTriggers().forEach(function(t) {
      if (t.getHandlerFunction() === 'processarChunkBulk') ScriptApp.deleteTrigger(t);
    });
    return;
  }

  var lista = JSON.parse(_getProp('bulk_lista') || '[]');
  var planIdx = parseInt(_getProp('bulk_planIdx') || '0');
  var fromRow = parseInt(_getProp('bulk_fromRow') || '2');
  var chunkGlobal = parseInt(_getProp('bulk_chunkGlobal') || '0');
  var totalLinhasAcm = parseInt(_getProp('bulk_totalLinhas') || '0');

  if (planIdx >= lista.length) {
    _log('🎉 Upload total CONCLUÍDO! ' + chunkGlobal + ' chunks | ' + totalLinhasAcm + ' linhas totais → Firebase');
    _setProp('bulk_ativo', '0');
    ScriptApp.getProjectTriggers().forEach(function(t) {
      if (t.getHandlerFunction() === 'processarChunkBulk') ScriptApp.deleteTrigger(t);
    });
    return;
  }

  var arq = lista[planIdx];
  _log('Processando ' + arq.nome + ' (plan ' + (planIdx+1) + '/' + lista.length + ', linha ' + fromRow + ')');

  var token = _getToken();
  if (!token) { _log('❌ Token Firebase falhou. Tentará novamente no próximo minuto.'); return; }

  try {
    var ss = SpreadsheetApp.openById(arq.id);
    var aba = ss.getSheets()[0];
    var ul = aba.getLastRow();
    var uc = aba.getLastColumn();

    if (ul < 2 || uc < 1) {
      _log('  (vazia) → avança para próxima');
      _setProp('bulk_planIdx', String(planIdx + 1));
      _setProp('bulk_fromRow', '2');
      return;
    }

    var cab = aba.getRange(1, 1, 1, uc).getValues()[0].map(function(v) { return String(v).trim(); });

    if (fromRow > ul) {
      // Planilha concluída
      var pct = Math.round((planIdx+1) / lista.length * 100);
      _log('  ✅ ' + arq.nome + ' completa! (' + pct + '% do total)');
      _setProp('bulk_planIdx', String(planIdx + 1));
      _setProp('bulk_fromRow', '2');
      return;
    }

    var end = Math.min(fromRow + CHUNK_LINHAS - 1, ul);
    var dados = aba.getRange(fromRow, 1, end - fromRow + 1, uc).getValues();
    var linhas = _parseLinhas(dados);

    if (!linhas.length) {
      _setProp('bulk_fromRow', String(end + 1));
      return;
    }

    // Nome do doc Firestore: "bulk/{NOME_PLANILHA}/chunk_{XXXX}"
    var nomeDoc = arq.nome.replace(/[^a-zA-Z0-9_\-]/g, '_');
    var totalChunksEst = Math.ceil((ul - 1) / CHUNK_LINHAS);
    var chunkLocal = Math.ceil((fromRow - 1) / CHUNK_LINHAS);
    var docPath = 'snapshots_bulk/' + nomeDoc + '/chunks/' + String(chunkLocal).padStart(4, '0');

    var ok = _fsWriteChunk(docPath, cab, linhas, chunkLocal, totalChunksEst, token);

    if (ok) {
      chunkGlobal++;
      totalLinhasAcm += linhas.length;
      _setProp('bulk_chunkGlobal', String(chunkGlobal));
      _setProp('bulk_totalLinhas', String(totalLinhasAcm));
      _setProp('bulk_fromRow', String(end + 1));
      _log('  chunk ' + chunkLocal + '/' + totalChunksEst + ' → ' + linhas.length + ' linhas ✅ (total acm: ' + totalLinhasAcm + ')');
    } else {
      _log('  ⚠️ Falha no chunk — tentará novamente no próximo minuto');
    }
  } catch(ex) {
    _log('  ❌ Erro: ' + ex);
  }
}

// ── VERIFICAR STATUS ────────────────────────────────────────────
function verificarStatusUpload() {
  var ativo = _getProp('bulk_ativo');
  var lista = JSON.parse(_getProp('bulk_lista') || '[]');
  var planIdx = parseInt(_getProp('bulk_planIdx') || '0');
  var chunkG = _getProp('bulk_chunkGlobal') || '0';
  var totalL = _getProp('bulk_totalLinhas') || '0';

  _log('═══════════════════════════════════');
  _log('STATUS DO UPLOAD TOTAL');
  _log('Ativo: ' + (ativo === '1' ? 'SIM' : 'NÃO (concluído ou não iniciado)'));
  _log('Planilhas: ' + planIdx + '/' + lista.length);
  _log('Chunks enviados: ' + chunkG);
  _log('Linhas totais: ' + totalL);
  if (planIdx < lista.length) _log('Próxima planilha: ' + lista[planIdx].nome);
  _log('═══════════════════════════════════');
}

// ── PARAR UPLOAD ────────────────────────────────────────────────
function pararUpload() {
  _setProp('bulk_ativo', '0');
  ScriptApp.getProjectTriggers().forEach(function(t) {
    if (t.getHandlerFunction() === 'processarChunkBulk') ScriptApp.deleteTrigger(t);
  });
  _log('Upload pausado. Execute iniciarUploadTotal() para recomeçar do zero, ou reativarUpload() para continuar.');
}

function reativarUpload() {
  _setProp('bulk_ativo', '1');
  ScriptApp.newTrigger('processarChunkBulk').timeBased().everyMinutes(1).create();
  _log('Upload reativado. Continuando de onde parou...');
}
